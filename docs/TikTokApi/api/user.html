134    def videos(self, count=30, cursor=0, **kwargs) -> Iterator[Video]:
135        """
136        Returns an iterator yielding Video objects.
137
138        - Parameters:
139            - count (int): The amount of videos you want returned.
140            - cursor (int): The unix epoch to get uploaded videos since.
141
142        Example Usage
143        ```py
144        user = api.user(username='therock')
145        for video in user.videos(count=100):
146            # do something
147        ```
148        """
149        processed = User.parent._process_kwargs(kwargs)
150        kwargs["custom_device_id"] = processed.device_id
151
152        if not self.user_id and not self.sec_uid:
153            self.__find_attributes()
154
155        first = True
156        amount_yielded = 0
157
158        while amount_yielded < count:
159            query = {
160                "count": 30,
161                "id": self.user_id,
162                "cursor": cursor,
163                "type": 1,
164                "secUid": self.sec_uid,
165                "sourceType": 8,
166                "appId": 1233,
167                "region": processed.region,
168                "priority_region": processed.region,
169                "language": processed.language,
170            }
171            path = "api/post/item_list/?{}&{}".format(
172                User.parent._add_url_params(), urlencode(query)
173            )
174
175            res = User.parent.get_data(path, send_tt_params=True, **kwargs)
176
177            videos = res.get("itemList", [])
178            for video in videos:
179                amount_yielded += 1
180                yield self.parent.video(data=video)
181
182            if not res.get("hasMore", False) and not first:
183                User.parent.logger.info(
184                    "TikTok isn't sending more TikToks beyond this point."
185                )
186                return
187
188            cursor = res["cursor"]
189            first = False

254    def __extract_from_data(self):
255        data = self.as_dict
256        keys = data.keys()
257
258        if "user_info" in keys:
259            self.__update_id_sec_uid_username(
260                data["user_info"]["uid"],
261                data["user_info"]["sec_uid"],
262                data["user_info"]["unique_id"],
263            )
264        elif "uniqueId" in keys:
265            self.__update_id_sec_uid_username(
266                data["id"], data["secUid"], data["uniqueId"]
267            )
268
269        if None in (self.username, self.user_id, self.sec_uid):
270            User.parent.logger.error(
271                f"Failed to create User with data: {data}\nwhich has keys {data.keys()}"
272            )

275        self.user_id = id

277        self.username = username

279    def __find_attributes(self) -> None:
280        # It is more efficient to check search first, since self.user_object() makes HTML request.
281        found = False
282        for u in self.parent.search.users(self.username):
283            if u.username == self.username:
284                found = True
285                self.__update_id_sec_uid_username(u.user_id, u.sec_uid, u.username)
286                break
287
288        if not found:
289            user_object = self.info()
290            self.__update_id_sec_uid_username(
291                user_object["id"], user_object["secUid"], user_object["uniqueId"]
292            )
