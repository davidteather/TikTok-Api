424    def external_signer(self, url, custom_device_id=None, verifyFp=None):
425        """Makes requests to an external signer instead of using a browser.
426
427        ##### Parameters
428        * url: The server to make requests to
429            This server is designed to sign requests. You can find an example
430            of this signature server in the examples folder.
431
432        * custom_device_id: A TikTok parameter needed to download videos

441        if custom_device_id is not None:
